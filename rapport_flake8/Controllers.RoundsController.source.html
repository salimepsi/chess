31 """

34 Args:

36  
37 Returns:
38 list: Liste des joueurs triés par score.
39 """
40 scores = set(player['score'] for player in players)
41 sorted_players = []
42 for score in sorted(scores, reverse=True):
43 players_with_same_score = [p for p in players if p['score'] == score]
